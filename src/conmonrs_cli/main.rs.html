1#![allow(clippy::needless_return)]
2use capnp_rpc::{RpcSystem, rpc_twoparty_capnp, twoparty};
3use conmon_common::conmon_capnp::conmon;
4use futures::{AsyncReadExt, FutureExt};
5use tokio::net::UnixStream;
6use tokio_util::compat::TokioAsyncReadCompatExt;
7
8#[tokio::main(flavor = "current_thread")]
9async fn main() -> Result<(), Box<dyn std::error::Error>> {
10    tokio::task::LocalSet::new()
11        .run_until(async move {
12            let stream = UnixStream::connect("conmon.sock").await?.compat();
13            let (reader, writer) = stream.split();
14
15            let rpc_network = Box::new(twoparty::VatNetwork::new(
16                reader,
17                writer,
18                rpc_twoparty_capnp::Side::Client,
19                Default::default(),
20            ));
21
22            let mut rpc_system = RpcSystem::new(rpc_network, None);
23            let client: conmon::Client = rpc_system.bootstrap(rpc_twoparty_capnp::Side::Server);
24
25            tokio::task::spawn_local(Box::pin(rpc_system.map(|_| ())));
26
27            let request = client.version_request();
28            let response = request.send().promise.await?;
29
30            println!(
31                "received: {}",
32                response.get()?.get_response()?.get_version()?.to_str()?
33            );
34            Ok(())
35        })
36        .await
37}
