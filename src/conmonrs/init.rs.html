14#[derive(Debug, Default)]
15pub struct Init<T> {
16    imp: T,
17}
18
19impl<T> Init<T>
20where
21    T: InitImpl,
22{
23    /// Unset the locale for the current process.
24    pub fn unset_locale(&self) -> Result<()> {
25        self.imp.setlocale(LC_ALL, CString::new("")?.as_ptr());
26        Ok(())
27    }
28
29    /// Helper to adjust the OOM score of the currently running process.
30    pub fn set_oom_score<S: AsRef<str>>(&self, score: S) -> Result<()> {
31        // Attempt adjustment with best-effort.
32        let mut file = self.imp.create_file("/proc/self/oom_score_adj")?;
33        if let Err(err) = self
34            .imp
35            .write_all_file(&mut file, score.as_ref().as_bytes())
36        {
37            match err.kind() {
38                ErrorKind::PermissionDenied => {
39                    println!("Missing sufficient privileges to adjust OOM score")
40                }
41                _ => bail!("adjusting OOM score {}", err),
42            }
43        }
44        Ok(())
45    }
46
47    /// Set the default umask
48    pub fn set_default_umask(&self) {
49        self.imp.umask(Mode::from_bits_truncate(0o022))
50    }
51}

53#[cfg_attr(test, automock)]
54pub trait InitImpl {
55    fn setlocale(&self, category: c_int, locale: *const c_char) -> *mut c_char;
56    fn create_file<P: 'static + AsRef<Path>>(&self, path: P) -> io::Result<File>;
57    fn write_all_file(&self, file: &mut File, buf: &[u8]) -> io::Result<()>;
58    fn umask(&self, mode: Mode);
59}
60
61#[derive(Debug, Default)]
62pub struct DefaultInit;
63
64impl InitImpl for DefaultInit {
65    fn setlocale(&self, category: c_int, locale: *const c_char) -> *mut c_char {
66        unsafe { setlocale(category, locale) }
67    }
68
69    fn create_file<P: AsRef<Path>>(&self, path: P) -> io::Result<File> {
70        File::create(path)
71    }
72
73    fn write_all_file(&self, file: &mut File, buf: &[u8]) -> io::Result<()> {
74        file.write_all(buf)
75    }
76
77    fn umask(&self, mode: Mode) {
78        umask(mode);
79    }
80}
