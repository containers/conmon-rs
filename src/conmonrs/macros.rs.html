1/// Lock a mutex and return a possible poison error converted to an `anyhow::Error`.
2/// ```ignore
3/// let mutex = Mutex::new(vec![]);
4/// // needs to be inside of a function that returns an anyhow::Result
5/// lock!(mutex).push(42);
6/// assert_eq!(lock!(mutex).pop(), Some(42));
7/// ```
8macro_rules! lock {
9    ($mutex:expr_2021) => {
10        $mutex.lock().map_err(|e| anyhow::anyhow!("{e}"))?
11    };
12}
