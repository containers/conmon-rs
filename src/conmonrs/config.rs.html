146#[derive(Clone, Debug, Deserialize, Eq, PartialEq, Serialize, Subcommand)]
147/// Possible subcommands.
148pub enum Commands {
149    /// Run pause, which bind mounts selected namespaces to the local file system.
150    ///
151    /// If a namespace is not selected by one of the flags, then it will fallback to the host
152    /// namespace and still create the bind mount to it. All namespaces are mounted to
153    /// /var/run/[ipc,pid,net,user,uts]ns/$POD_ID, whereas the POD_ID is being passed from the
154    /// client.
155    ///
156    /// Tracking of the pause PID will be done by using a file in /var/run/conmonrs/$POD_ID.pid,
157    /// which gets removed together with the mounted namespaces if `conmonrs pause` terminates.
158    ///
159    /// UID and GID mappings are required if unsharing of the user namespace (via `--user`) is
160    /// selected.
161    Pause {
162        #[arg(
163            default_value("/var/run"),
164            env(concat!(prefix!(), "PAUSE_BASE_PATH")),
165            long("base-path"),
166            short('p'),
167            value_name("PATH")
168        )]
169        /// The base path for pinning the namespaces.
170        base_path: PathBuf,
171
172        #[arg(
173            env(concat!(prefix!(), "PAUSE_POD_ID")),
174            long("pod-id"),
175        )]
176        /// The unique pod identifier for referring to the namespaces.
177        pod_id: String,
178
179        #[arg(long("ipc"))]
180        /// Unshare the IPC namespace.
181        ipc: bool,
182
183        #[arg(long("pid"))]
184        /// Unshare the PID namespace.
185        pid: bool,
186
187        #[arg(long("net"))]
188        /// Unshare the network namespace.
189        net: bool,
190
191        #[arg(long("user"))]
192        /// Unshare the user namespace.
193        user: bool,
194
195        #[arg(long("uts"))]
196        /// Unshare the UTS namespace.
197        uts: bool,
198
199        #[arg(long("uid-mappings"), required_if_eq("user", "true"), short('u'))]
200        /// User ID mappings for unsahring the user namespace.
201        /// Allows multiple mappings in the format: "CONTAINER_ID HOST_ID SIZE".
202        uid_mappings: Vec<String>,
203
204        #[arg(long("gid-mappings"), required_if_eq("user", "true"), short('g'))]
205        /// Group ID mappings for unsahring the user namespace.
206        /// Allows multiple mappings in the format: "CONTAINER_ID HOST_ID SIZE".
207        gid_mappings: Vec<String>,
208    },
209}

314// Sync with `pkg/client/client.go`
315const SOCKET: &str = "conmon.sock";
316const PIDFILE: &str = "pidfile";
317const FD_SOCKET: &str = "conmon-fd.sock";
318
319impl Config {
320    /// Validate the configuration integrity.
321    pub fn validate(&self) -> Result<()> {
322        const RUNTIME_FLAG: &str = "--runtime";
323        if self
324            .runtime()
325            .to_str()
326            .context(format!("{RUNTIME_FLAG} does not parse as string"))?
327            .trim()
328            .is_empty()
329        {
330            bail!("{} flag not set", RUNTIME_FLAG)
331        }
332
333        const RUNTIME_DIR_FLAG: &str = "--runtime-dir";
334        if self
335            .runtime_dir()
336            .to_str()
337            .context(format!("{RUNTIME_DIR_FLAG} does not parse as string"))?
338            .trim()
339            .is_empty()
340        {
341            bail!("{} flag not set", RUNTIME_DIR_FLAG)
342        }
343
344        if !self.runtime().exists() {
345            bail!(
346                "{} '{}' does not exist",
347                RUNTIME_FLAG,
348                self.runtime().display()
349            )
350        }
351
352        const RUNTIME_ROOT_FLAG: &str = "--runtime-root";
353        if !self.runtime_dir().exists() {
354            fs::create_dir_all(self.runtime_dir())?;
355        }
356
357        if let Some(rr) = self.runtime_root() {
358            if !rr.exists() {
359                fs::create_dir_all(rr)?;
360            } else if !rr.is_dir() {
361                bail!("{} '{}' does not exist", RUNTIME_ROOT_FLAG, rr.display())
362            }
363        }
364
365        if self.socket().exists() {
366            fs::remove_file(self.socket())?;
367        }
368
369        if self.fd_socket().exists() {
370            fs::remove_file(self.fd_socket())?;
371        }
372
373        Ok(())
374    }
375    pub fn socket(&self) -> PathBuf {
376        self.runtime_dir().join(SOCKET)
377    }
378    pub fn conmon_pidfile(&self) -> PathBuf {
379        self.runtime_dir().join(PIDFILE)
380    }
381    pub fn fd_socket(&self) -> PathBuf {
382        self.runtime_dir().join(FD_SOCKET)
383    }
