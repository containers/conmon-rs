16/// The main structure of this module.
17pub struct Telemetry;
18
19impl Telemetry {
20    /// Return the telemetry layer if tracing is enabled.
21    pub fn layer(endpoint: &str) -> Result<SdkTracerProvider> {
22        global::set_text_map_propagator(TraceContextPropagator::new());
23
24        let exporter = SpanExporter::builder()
25            .with_tonic()
26            .with_endpoint(endpoint)
27            .build()
28            .context("build exporter")?;
29
30        let hostname = gethostname()
31            .context("get hostname")?
32            .to_str()
33            .context("convert hostname to string")?
34            .to_string();
35
36        let tracer = SdkTracerProvider::builder()
37            .with_batch_exporter(exporter)
38            .with_resource(
39                Resource::builder()
40                    .with_service_name(crate_name!())
41                    .with_attributes([
42                        KeyValue::new(PROCESS_PID, process::id() as i64),
43                        KeyValue::new(HOST_NAME, hostname),
44                    ])
45                    .build(),
46            )
47            .build();
48
49        Ok(tracer)
50    }
51
52    /// Set a new parent context from the provided slice data.
53    pub fn set_parent_context(reader: Reader<conmon::text_text_map_entry::Owned>) -> Result<()> {
54        if reader.is_empty() {
55            // Make it a noop if no data is provided.
56            return Ok(());
57        }
58
59        let metadata = Metadata(capnp_util::into_map(reader)?);
60        let ctx = global::get_text_map_propagator(|prop| prop.extract(&metadata));
61        let _ = Span::current().set_parent(ctx);
62
63        Ok(())
64    }
65}

67/// Additional telemetry metadata to carry.
68struct Metadata<'a>(HashMap<&'a str, &'a str>);
69
70impl Extractor for Metadata<'_> {
71    fn get(&self, key: &str) -> Option<&str> {
72        self.0.get(key).copied()
73    }
74
75    /// Collect all the keys from the MetadataMap.
76    fn keys(&self) -> Vec<&str> {
77        self.0.keys().copied().collect()
78    }
79}
