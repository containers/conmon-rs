121impl ErrorMessage {
122    fn new<T>(exit_code: T) -> Self
123    where
124        T: ToString,
125    {
126        Self {
127            status: "Failure",
128            reason: "NonZeroExitCode",
129            details: ErrorDetails {
130                causes: vec![ErrorCause {

133                }],
134            },
135            message: "command terminated with non-zero exit code",

137    }

154    /// Start the streaming server if not already running.
155    pub async fn start_if_required(&mut self) -> Result<()> {
156        if self.running {
157            return Ok(());
158        }
159
160        let listener = TcpListener::bind(ADDR.to_string() + ":0")
161            .await
162            .context("bind streaming server")?;
163
164        let local_addr = listener
165            .local_addr()
166            .context("get listeners local address")?;
167
168        self.port = local_addr.port();
169
170        info!("Starting streaming server on {local_addr}");
171        task::spawn_local(Self::serve(listener, self.state.clone()));
172        self.running = true;
173
174        Ok(())
175    }

177    /// Serve the main streaming server.
178    async fn serve(listener: TcpListener, state: Arc<RwLock<State>>) -> Result<()> {
179        let router = Router::new()
180            .route(&Self::path_for(EXEC_PATH), get(Self::handle))
181            .route(&Self::path_for(ATTACH_PATH), get(Self::handle))
182            .route(&Self::path_for(PORT_FORWARD_PATH), get(Self::handle))
183            // for allowing proper error handling on wrong protocol usage (spdy instead of websocket)
184            .route(&Self::path_for(EXEC_PATH), post(Self::handle))
185            .route(&Self::path_for(ATTACH_PATH), post(Self::handle))
186            .route(&Self::path_for(PORT_FORWARD_PATH), post(Self::handle))
187            .fallback(Self::fallback)
188            .with_state(state)
189            .layer(
190                TraceLayer::new_for_http()
191                    .make_span_with(DefaultMakeSpan::default().include_headers(true)),
192            );
193        axum::serve(listener, router)
194            .await
195            .context("start streaming server")
196    }

282    /// Handle a webserver connection which should be upgraded to become a websocket one.
283    async fn handle(
284        ws: WebSocketUpgrade,
285        Path(token): Path<Uuid>,
286        AxumState(state): AxumState<Arc<RwLock<State>>>,
287    ) -> impl IntoResponse {
288        let span = debug_span!("handle_common", token = token.to_string().as_str());
289        let _enter = span.enter();
290
291        info!("Got request for token: {token}");
292        let mut state_lock = state.write().await;
293
294        match state_lock.remove(&token) {
295            Some(session) => {
296                info!("Got valid session for token {token}: {session:?}");
297                ws.protocols([PROTOCOL_V5, PROTOCOL_PORT_FORWARD])
298                    .on_upgrade(move |socket| Self::handle_websocket(socket, session))
299            }
300            None => {
301                error!("Unable to find session for token: {token}");
302                StatusCode::NOT_FOUND.into_response()
303            }
304        }
305    }

307    /// Handle a single websocket connection.
308    async fn handle_websocket(socket: WebSocket, session: Session) {
309        let (sender, receiver) = socket.split();
310        let (stdin_tx, stdin_rx) = unbounded_channel();
311
312        let mut send_task = Self::write_task(sender, stdin_rx, session).await;
313        let mut recv_task = Self::read_task(receiver, stdin_tx).await;
314
315        tokio::select! {
316            rv_a = (&mut send_task) => {
317                match rv_a {
318                    Ok(_) => info!("All messages sent"),
319                    Err(a) => error!("Error sending messages: {a:?}")
320                }
321                recv_task.abort();
322            },
323            rv_b = (&mut recv_task) => {
324                match rv_b {
325                    Ok(_) => info!("All messages received"),
326                    Err(b) => error!("Error receiving messages: {b:?}")
327                }
328                send_task.abort();
329            }
330        }
331
332        info!("Closing websocket connection");
333    }

335    /// Build a common write task based on the session type.
336    async fn write_task(
337        mut sender: SplitSink<WebSocket, Message>,
338        stdin_rx: UnboundedReceiver<Vec<u8>>,
339        session: Session,
340    ) -> JoinHandle<()> {
341        tokio::spawn(async move {
342            if let (Err(e), typ) = match session {
343                Session::Exec(s) => (Self::exec_loop(*s, &mut sender, stdin_rx).await, "exec"),
344                Session::Attach(s) => (Self::attach_loop(s, &mut sender, stdin_rx).await, "attach"),
345                Session::PortForward(s) => (
346                    Self::port_forward_loop(s, &mut sender, stdin_rx).await,
347                    "port forward",
348                ),
349            } {
350                error!("Unable to run {typ} for container: {e}");
351            }
352
353            if let Err(e) = sender
354                .send(Message::Close(
355                    CloseFrame {
356                        code: close_code::NORMAL,
357                        reason: "done".into(),
358                    }
359                    .into(),
360                ))
361                .await
362            {
363                error!("Unable to send close message: {e}")
364            }
365        })
366    }

368    /// Build a common read task.
369    async fn read_task(
370        mut receiver: SplitStream<WebSocket>,
371        mut stdin_tx: UnboundedSender<Vec<u8>>,
372    ) -> JoinHandle<()> {
373        tokio::spawn(async move {
374            while let Some(Ok(msg)) = receiver.next().await {
375                if Self::read_message(msg, &mut stdin_tx).is_break() {
376                    break;
377                }
378            }
379        })
380    }

382    /// Read a single message and return the control flow decision.
383    fn read_message(msg: Message, stdin_tx: &mut UnboundedSender<Vec<u8>>) -> ControlFlow<(), ()> {
384        match msg {
385            Message::Binary(data) if !data.is_empty() => {
386                debug!("Got {} binary bytes", data.len());
387                if let Err(e) = stdin_tx.send(data.into()) {
388                    error!("Unable to send stdin data: {e}");
389                }
390            }
391            Message::Close(c) => {
392                if let Some(cf) = c {
393                    info!(
394                        "Got websocket close with code {} and reason `{}`",
395                        cf.code, cf.reason
396                    );
397                } else {
398                    warn!("Got close message without close frame");
399                }
400                return ControlFlow::Break(());
401            }
402            Message::Text(t) => trace!("Got text message: {t:?}"),
403            Message::Pong(_) => trace!("Got pong"),
404            Message::Ping(_) => trace!("Got ping"),
405            Message::Binary(_) => trace!("Got unknown binary data"),
406        }
407
408        ControlFlow::Continue(())
409    }

411    /// The exec specific read/write loop.
412    async fn exec_loop(
413        mut session: ExecSession,
414        sender: &mut SplitSink<WebSocket, Message>,
415        mut stdin_rx: UnboundedReceiver<Vec<u8>>,
416    ) -> Result<()> {
417        let pidfile = ContainerIO::temp_file_name(
418            Some(session.server_config.runtime_dir()),
419            "exec_streaming",
420            "pid",
421        )
422        .context("build pid file path")?;
423
424        let args = GenerateRuntimeArgs {
425            config: &session.server_config,
426            id: &session.container_id,
427            container_io: &session.container_io,
428            pidfile: &pidfile,
429            cgroup_manager: session.cgroup_manager,
430        };
431        let mut args = args
432            .exec_sync_args_without_command()
433            .context("exec sync args without command")?;
434        args.extend(session.command);
435
436        let (grandchild_pid, token) = session
437            .child_reaper
438            .create_child(
439                session.server_config.runtime(),
440                &args,
441                session.stdin,
442                &mut session.container_io,
443                &pidfile,
444                vec![],
445                vec![],
446            )
447            .await
448            .context("create new child process")?;
449
450        let io = SharedContainerIO::new(session.container_io);
451        let child = Child::new(
452            session.container_id.clone(),
453            grandchild_pid,
454            vec![],
455            vec![],
456            None,
457            io.clone(),
458            vec![],
459            token.clone(),
460        );
461
462        let mut exit_rx = session
463            .child_reaper
464            .watch_grandchild(child, vec![])
465            .context("watch grandchild for pid")?;
466
467        let (stdout_rx, stderr_rx) = io
468            .stdio()
469            .await
470            .context("retrieve stdout and stderr channels")?;
471
472        loop {
473            tokio::select! {
474                Some(mut data) = stdin_rx.recv()  => if session.stdin {
475                    // First element is the message type indicator
476                    match data.remove(0) {
477                        STDIN_BYTE => {
478                            trace!("Got stdin message of len {}", data.len());
479                            io.attach().await.stdin().send(data).context("send to attach session")?;
480                        },
481                        RESIZE_BYTE => {
482                            let e = serde_json::from_slice::<ResizeEvent>(&data).context("unmarshal resize event")?;
483                            trace!("Got resize message: {e:?}");
484                            io.resize(e.width, e.height).await.context("resize terminal")?;
485                        },
486                        CLOSE_BYTE => {
487                            info!("Got close message");
488                            break
489                        },
490                        x => warn!("Unknown start byte for stdin: {x}"),
491                    };
492                },
493
494                Ok(IOMessage::Data(mut data, _)) = stdout_rx.recv() => if session.stdout {
495                    data.insert(0, STDOUT_BYTE);
496                    sender.send(Message::Binary(data.into())).await.context("send to stdout")?;
497                },
498
499                Ok(IOMessage::Data(mut data, _)) = stderr_rx.recv() => if session.stderr {
500                    data.insert(0, STDERR_BYTE);
501                    sender.send(Message::Binary(data.into())).await.context("send to stderr")?;
502                },
503
504                Ok(exit_data) = exit_rx.recv() => {
505                    if exit_data.exit_code != 0 {
506                        let mut err = vec![STREAM_ERR_BYTE];
507                        let msg = ErrorMessage::new(exit_data.exit_code);
508                        err.extend(serde_json::to_vec(&msg).context("serialize error message")?);
509                        sender.send(Message::Binary(err.into())).await.context("send exit failure message")?;
510                    }
511                    break
512                },
513            }
514        }
515
516        Ok(())
517    }

519    /// The attach specific read/write loop.
520    async fn attach_loop(
521        session: AttachSession,
522        sender: &mut SplitSink<WebSocket, Message>,
523        mut stdin_rx: UnboundedReceiver<Vec<u8>>,
524    ) -> Result<()> {
525        let io = session.child.io();
526
527        let (stdout_rx, stderr_rx) = io
528            .stdio()
529            .await
530            .context("retrieve stdout and stderr channels")?;
531
532        loop {
533            tokio::select! {
534                Some(mut data) = stdin_rx.recv()  => if session.stdin {
535                    // First element is the message type indicator
536                    match data.remove(0) {
537                        STDIN_BYTE => {
538                            trace!("Got stdin message of len {}", data.len());
539                            io.attach().await.stdin().send(data).context("send to attach session")?;
540                        },
541                        RESIZE_BYTE => {
542                            let e = serde_json::from_slice::<ResizeEvent>(&data).context("unmarshal resize event")?;
543                            trace!("Got resize message: {e:?}");
544                            io.resize(e.width, e.height).await.context("resize terminal")?;
545                        },
546                        CLOSE_BYTE => {
547                            info!("Got close message");
548                            break
549                        },
550                        x => warn!("Unknown start byte for stdin: {x}"),
551                    };
552                },
553
554                Ok(IOMessage::Data(mut data, _)) = stdout_rx.recv() => if session.stdout {
555                    data.insert(0, STDOUT_BYTE);
556                    sender.send(Message::Binary(data.into())).await.context("send to stdout")?;
557                },
558
559                Ok(IOMessage::Data(mut data, _)) = stderr_rx.recv() => if session.stderr {
560                    data.insert(0, STDERR_BYTE);
561                    sender.send(Message::Binary(data.into())).await.context("send to stderr")?;
562                },
563
564                _ = session.child.token().cancelled() => {
565                    debug!("Exiting streaming attach because token cancelled");
566                    break
567                }
568            }
569        }
570
571        Ok(())
572    }
