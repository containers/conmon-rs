113/// Error message type used for exec in case that the command fails.

115    status: &'static str,
116    reason: &'static str,
117    details: ErrorDetails,

119}

121impl ErrorMessage {
122    fn new<T>(exit_code: T) -> Self
123    where
124        T: ToString,
125    {
126        Self {
127            status: "Failure",
128            reason: "NonZeroExitCode",
129            details: ErrorDetails {
130                causes: vec![ErrorCause {
131                    reason: "ExitCode",
132                    message: exit_code.to_string(),
133                }],
134            },
135            message: "command terminated with non-zero exit code",
136        }
137    }
138}

154    /// Start the streaming server if not already running.
155    pub async fn start_if_required(&mut self) -> Result<()> {
156        if self.running {
157            return Ok(());
158        }
159
160        let listener = TcpListener::bind(ADDR.to_string() + ":0")
161            .await
162            .context("bind streaming server")?;
163
164        let local_addr = listener
165            .local_addr()
166            .context("get listeners local address")?;
167
168        self.port = local_addr.port();
169
170        info!("Starting streaming server on {local_addr}");
171        task::spawn_local(Self::serve(listener, self.state.clone()));
172        self.running = true;
173
174        Ok(())
175    }

177    /// Serve the main streaming server.
178    async fn serve(listener: TcpListener, state: Arc<RwLock<State>>) -> Result<()> {
179        let router = Router::new()
180            .route(&Self::path_for(EXEC_PATH), get(Self::handle))
181            .route(&Self::path_for(ATTACH_PATH), get(Self::handle))
182            .route(&Self::path_for(PORT_FORWARD_PATH), get(Self::handle))
183            // for allowing proper error handling on wrong protocol usage (spdy instead of websocket)
184            .route(&Self::path_for(EXEC_PATH), post(Self::handle))
185            .route(&Self::path_for(ATTACH_PATH), post(Self::handle))
186            .route(&Self::path_for(PORT_FORWARD_PATH), post(Self::handle))
187            .fallback(Self::fallback)
188            .with_state(state)
189            .layer(TraceLayer::new_for_http());
190        axum::serve(listener, router)
191            .await
192            .context("start streaming server")
193    }

281    /// Handle a webserver connection which should be upgraded to become a websocket one.
282    async fn handle(
283        ws: WebSocketUpgrade,
284        Path(token): Path<Uuid>,
285        AxumState(state): AxumState<Arc<RwLock<State>>>,
286    ) -> impl IntoResponse {
287        let span = debug_span!("handle_common", %token);
288        let _enter = span.enter();
289
290        info!("Got request for token: {token}");
291        let mut state_lock = state.write().await;
292
293        match state_lock.remove(&token) {
294            Some(session) => {
295                info!("Got valid session for token {token}: {session:?}");
296                ws.protocols([PROTOCOL_V5, PROTOCOL_PORT_FORWARD])
297                    .on_upgrade(move |socket| Self::handle_websocket(socket, session))
298            }
299            None => {
300                error!("Unable to find session for token: {token}");
301                StatusCode::NOT_FOUND.into_response()
302            }
303        }
304    }

306    /// Handle a single websocket connection.
307    async fn handle_websocket(socket: WebSocket, session: Session) {
308        let (sender, receiver) = socket.split();
309        let (stdin_tx, stdin_rx) = mpsc::channel(16);
310
311        let mut send_task = Self::write_task(sender, stdin_rx, session).await;
312        let mut recv_task = Self::read_task(receiver, stdin_tx).await;
313
314        tokio::select! {
315            rv_a = (&mut send_task) => {
316                match rv_a {
317                    Ok(_) => info!("All messages sent"),
318                    Err(a) => error!("Error sending messages: {a:?}")
319                }
320                recv_task.abort();
321            },
322            rv_b = (&mut recv_task) => {
323                match rv_b {
324                    Ok(_) => info!("All messages received"),
325                    Err(b) => error!("Error receiving messages: {b:?}")
326                }
327                send_task.abort();
328            }
329        }
330
331        info!("Closing websocket connection");
332    }

334    /// Build a common write task based on the session type.
335    async fn write_task(
336        mut sender: SplitSink<WebSocket, Message>,
337        stdin_rx: MpscReceiver<Vec<u8>>,
338        session: Session,
339    ) -> JoinHandle<()> {
340        tokio::spawn(async move {
341            if let (Err(e), typ) = match session {
342                Session::Exec(s) => (Self::exec_loop(*s, &mut sender, stdin_rx).await, "exec"),
343                Session::Attach(s) => (Self::attach_loop(s, &mut sender, stdin_rx).await, "attach"),
344                Session::PortForward(s) => (
345                    Self::port_forward_loop(s, &mut sender, stdin_rx).await,
346                    "port forward",
347                ),
348            } {
349                error!("Unable to run {typ} for container: {e}");
350            }
351
352            if let Err(e) = sender
353                .send(Message::Close(
354                    CloseFrame {
355                        code: close_code::NORMAL,
356                        reason: "done".into(),
357                    }
358                    .into(),
359                ))
360                .await
361            {
362                error!("Unable to send close message: {e}")
363            }
364        })
365    }

367    /// Build a common read task.
368    async fn read_task(
369        mut receiver: SplitStream<WebSocket>,
370        stdin_tx: MpscSender<Vec<u8>>,
371    ) -> JoinHandle<()> {
372        tokio::spawn(async move {
373            while let Some(Ok(msg)) = receiver.next().await {
374                if Self::read_message(msg, &stdin_tx).await.is_break() {
375                    break;
376                }
377            }
378        })
379    }

381    /// Read a single message and return the control flow decision.
382    async fn read_message(msg: Message, stdin_tx: &MpscSender<Vec<u8>>) -> ControlFlow<(), ()> {
383        match msg {
384            Message::Binary(data) if !data.is_empty() => {
385                debug!("Got {} binary bytes", data.len());
386                if let Err(e) = stdin_tx.send(data.into()).await {
387                    error!("Unable to send stdin data: {e}");
388                }
389            }
390            Message::Close(c) => {
391                if let Some(cf) = c {
392                    info!(
393                        "Got websocket close with code {} and reason `{}`",
394                        cf.code, cf.reason
395                    );
396                } else {
397                    warn!("Got close message without close frame");
398                }
399                return ControlFlow::Break(());
400            }
401            Message::Text(t) => trace!("Got text message: {t:?}"),
402            Message::Pong(_) => trace!("Got pong"),
403            Message::Ping(_) => trace!("Got ping"),
404            Message::Binary(_) => trace!("Got unknown binary data"),
405        }
406
407        ControlFlow::Continue(())
408    }

410    /// The exec specific read/write loop.
411    async fn exec_loop(
412        mut session: ExecSession,
413        sender: &mut SplitSink<WebSocket, Message>,
414        mut stdin_rx: MpscReceiver<Vec<u8>>,
415    ) -> Result<()> {
416        let pidfile = ContainerIO::temp_file_name(
417            Some(session.server_config.runtime_dir()),
418            "exec_streaming",
419            "pid",
420        );
421
422        let args = GenerateRuntimeArgs {
423            config: &session.server_config,
424            id: &session.container_id,
425            container_io: &session.container_io,
426            pidfile: &pidfile,
427            cgroup_manager: session.cgroup_manager,
428        };
429        let mut args = args
430            .exec_sync_args_without_command()
431            .context("exec sync args without command")?;
432        args.extend(session.command);
433
434        let (grandchild_pid, token) = session
435            .child_reaper
436            .create_child(
437                session.server_config.runtime(),
438                &args,
439                session.stdin,
440                &mut session.container_io,
441                &pidfile,
442                vec![],
443                vec![],
444            )
445            .await
446            .context("create new child process")?;
447
448        let container_id = session.container_id;
449        let io = SharedContainerIO::new(session.container_io);
450        let child = Child::new(
451            container_id,
452            grandchild_pid,
453            vec![],
454            vec![],
455            None,
456            io.clone(),
457            vec![],
458            token.clone(),
459        );
460
461        let mut exit_rx = session
462            .child_reaper
463            .watch_grandchild(child, vec![])
464            .context("watch grandchild for pid")?;
465
466        let (stdout_rx, stderr_rx) = io
467            .stdio()
468            .await
469            .context("retrieve stdout and stderr channels")?;
470
471        let attach = io.attach().await;
472
473        loop {
474            tokio::select! {
475                Some(data) = stdin_rx.recv()  => if session.stdin {
476                    // First element is the message type indicator
477                    if let Some((&msg_type, payload)) = data.split_first() {
478                        match msg_type {
479                            STDIN_BYTE => {
480                                trace!("Got stdin message of len {}", payload.len());
481                                attach.stdin().send(Arc::from(payload)).context("send to attach session")?;
482                            },
483                            RESIZE_BYTE => {
484                                let e = serde_json::from_slice::<ResizeEvent>(payload).context("unmarshal resize event")?;
485                                trace!("Got resize message: {e:?}");
486                                io.resize(e.width, e.height).await.context("resize terminal")?;
487                            },
488                            CLOSE_BYTE => {
489                                info!("Got close message");
490                                break
491                            },
492                            x => warn!("Unknown start byte for stdin: {x}"),
493                        }
494                    }
495                },
496
497                Ok(IOMessage::Data(data, _)) = stdout_rx.recv() => if session.stdout {
498                    Self::frame_and_send(STDOUT_BYTE, &data, sender).await
499                        .context("send to stdout")?;
500                },
501
502                Ok(IOMessage::Data(data, _)) = stderr_rx.recv() => if session.stderr {
503                    Self::frame_and_send(STDERR_BYTE, &data, sender).await
504                        .context("send to stderr")?;
505                },
506
507                Ok(exit_data) = exit_rx.recv() => {
508                    if exit_data.exit_code != 0 {
509                        let mut err = vec![STREAM_ERR_BYTE];
510                        let msg = ErrorMessage::new(exit_data.exit_code);
511                        err.extend(serde_json::to_vec(&msg).context("serialize error message")?);
512                        sender.send(Message::Binary(err.into())).await.context("send exit failure message")?;
513                    }
514                    break
515                },
516            }
517        }
518
519        Ok(())
520    }

522    /// The attach specific read/write loop.
523    async fn attach_loop(
524        session: AttachSession,
525        sender: &mut SplitSink<WebSocket, Message>,
526        mut stdin_rx: MpscReceiver<Vec<u8>>,
527    ) -> Result<()> {
528        let io = session.child.io();
529
530        let (stdout_rx, stderr_rx) = io
531            .stdio()
532            .await
533            .context("retrieve stdout and stderr channels")?;
534
535        let attach = io.attach().await;
536
537        loop {
538            tokio::select! {
539                Some(data) = stdin_rx.recv()  => if session.stdin {
540                    // First element is the message type indicator
541                    if let Some((&msg_type, payload)) = data.split_first() {
542                        match msg_type {
543                            STDIN_BYTE => {
544                                trace!("Got stdin message of len {}", payload.len());
545                                attach.stdin().send(Arc::from(payload)).context("send to attach session")?;
546                            },
547                            RESIZE_BYTE => {
548                                let e = serde_json::from_slice::<ResizeEvent>(payload).context("unmarshal resize event")?;
549                                trace!("Got resize message: {e:?}");
550                                io.resize(e.width, e.height).await.context("resize terminal")?;
551                            },
552                            CLOSE_BYTE => {
553                                info!("Got close message");
554                                break
555                            },
556                            x => warn!("Unknown start byte for stdin: {x}"),
557                        }
558                    }
559                },
560
561                Ok(IOMessage::Data(data, _)) = stdout_rx.recv() => if session.stdout {
562                    Self::frame_and_send(STDOUT_BYTE, &data, sender).await
563                        .context("send to stdout")?;
564                },
565
566                Ok(IOMessage::Data(data, _)) = stderr_rx.recv() => if session.stderr {
567                    Self::frame_and_send(STDERR_BYTE, &data, sender).await
568                        .context("send to stderr")?;
569                },
570
571                _ = session.child.token().cancelled() => {
572                    debug!("Exiting streaming attach because token cancelled");
573                    break
574                }
575            }
576        }
577
578        Ok(())
579    }

590    /// Prepend a stream type byte and send the data over WebSocket.
591    async fn frame_and_send(
592        stream_byte: u8,
593        data: &[u8],
594        sender: &mut SplitSink<WebSocket, Message>,
595    ) -> Result<()> {
596        let mut framed = Vec::with_capacity(1 + data.len());
597        framed.push(stream_byte);
598        framed.extend_from_slice(data);
599        sender
600            .send(Message::Binary(framed.into()))
601            .await
602            .context("send framed data")
603    }
