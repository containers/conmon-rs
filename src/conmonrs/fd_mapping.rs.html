1//! File descriptor mapping for child process spawning.
2
3use nix::{
4    fcntl::{FcntlArg, FdFlag, fcntl},
5    unistd::dup2_raw,
6};
7use std::{
8    cmp::max,
9    fmt, io,
10    os::fd::{AsRawFd, FromRawFd, IntoRawFd, OwnedFd, RawFd},
11};
12use tokio::process::Command;
13
14/// A mapping from a file descriptor in the parent to a file descriptor in the child.
15#[derive(Debug)]
16pub struct FdMapping {
17    pub parent_fd: OwnedFd,
18    pub child_fd: RawFd,
19}
20
21/// Extension to add file descriptor mappings to a Command.
22pub trait CommandFdExt {
23    fn fd_mappings(&mut self, mappings: Vec<FdMapping>) -> Result<&mut Self, FdMappingCollision>;
24}
25
26/// Error when two or more mappings target the same child FD.
27#[derive(Copy, Clone, Debug, Eq, PartialEq)]
28pub struct FdMappingCollision;
29
30impl fmt::Display for FdMappingCollision {
31    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
32        write!(f, "Two or more mappings for the same child FD")
33    }
34}
35
36impl std::error::Error for FdMappingCollision {}
37
38impl CommandFdExt for Command {
39    fn fd_mappings(
40        &mut self,
41        mut mappings: Vec<FdMapping>,
42    ) -> Result<&mut Self, FdMappingCollision> {
43        let child_fds = validate_child_fds(&mappings)?;
44
45        // Safety: map_fds will not allocate, so it is safe to call from this hook.
46        unsafe {
47            self.pre_exec(move || map_fds(&mut mappings, &child_fds));
48        }
49
50        Ok(self)
51    }
52}
53
54fn validate_child_fds(mappings: &[FdMapping]) -> Result<Vec<RawFd>, FdMappingCollision> {
55    let mut child_fds: Vec<RawFd> = mappings.iter().map(|m| m.child_fd).collect();
56    child_fds.sort_unstable();
57    child_fds.dedup();
58    if child_fds.len() != mappings.len() {
59        return Err(FdMappingCollision);
60    }
61    Ok(child_fds)
62}
63
64// This function must not allocate, as it is called from the pre_exec hook.
65fn map_fds(mappings: &mut [FdMapping], child_fds: &[RawFd]) -> io::Result<()> {
66    if mappings.is_empty() {
67        return Ok(());
68    }
69
70    let first_safe_fd = mappings
71        .iter()
72        .map(|m| max(m.parent_fd.as_raw_fd(), m.child_fd))
73        .max()
74        .expect("mappings is non-empty")
75        + 1;
76
77    for mapping in mappings.iter_mut() {
78        if child_fds.contains(&mapping.parent_fd.as_raw_fd())
79            && mapping.parent_fd.as_raw_fd() != mapping.child_fd
80        {
81            let parent_fd = fcntl(&mapping.parent_fd, FcntlArg::F_DUPFD_CLOEXEC(first_safe_fd))?;
82            unsafe {
83                mapping.parent_fd = OwnedFd::from_raw_fd(parent_fd);
84            }
85        }
86    }
87
88    for mapping in mappings {
89        if mapping.child_fd == mapping.parent_fd.as_raw_fd() {
90            fcntl(&mapping.parent_fd, FcntlArg::F_SETFD(FdFlag::empty()))?;
91        } else {
92            unsafe {
93                let _ = dup2_raw(&mapping.parent_fd, mapping.child_fd)?.into_raw_fd();
94            }
95        }
96    }
97
98    Ok(())
99}
