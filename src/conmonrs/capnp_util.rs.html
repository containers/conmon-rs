6/// Helper functions to convert between capnp maps (List's of MapEntries) and rust maps
7mod map_entry {
8    use capnp::{
9        struct_list::{Builder, Reader},
10        traits::OwnedStruct,
11    };
12    use conmon_common::conmon_capnp::conmon;
13
14    /// Convert from a MapEntry capnp struct to a (key, value) tuple.
15    pub(crate) trait IntoKeyValue<'a, K, V>: OwnedStruct {
16        fn into_key_value(entry: Self::Reader<'a>) -> capnp::Result<(K, V)>;
17    }
18
19    /// Set the key and value properties of a MapEntry capnp struct.
20    pub(crate) trait SetKeyValue<K, V>: OwnedStruct {
21        fn set_key_value(entry: Self::Builder<'_>, key: K, value: V);
22    }
23
24    impl<'a, K, V> IntoKeyValue<'a, K, V> for conmon::text_text_map_entry::Owned
25    where
26        K: From<&'a str>,
27        V: From<&'a str>,
28    {
29        fn into_key_value(entry: Self::Reader<'a>) -> capnp::Result<(K, V)> {
30            Ok((
31                entry.get_key()?.to_str()?.into(),
32                entry.get_value()?.to_str()?.into(),
33            ))
34        }
35    }
36
37    impl<K, V> SetKeyValue<K, V> for conmon::text_text_map_entry::Owned
38    where
39        K: AsRef<str>,
40        V: AsRef<str>,
41    {
42        fn set_key_value(mut entry: Self::Builder<'_>, key: K, value: V) {
43            entry.set_key(key.as_ref());
44            entry.set_value(value.as_ref());
45        }
46    }
47
48    /// Convert a capnp map reader (`struct_list::Reader`) into a rust map (`impl FromIterator<(K, V)>`).
49    pub(crate) fn into_map<'a, K, V, E, T>(reader: Reader<'a, E>) -> capnp::Result<T>
50    where
51        E: IntoKeyValue<'a, K, V>,
52        T: FromIterator<(K, V)>,
53    {
54        reader.into_iter().map(E::into_key_value).collect()
55    }
56
57    /// Set a capnp map property from a rust map (`impl IntoIterator<Item=(K, V)>`).
58    ///
59    /// The `init` function is used to allocate a `struct_list::Builder`.
60    ///
61    /// ## Example
62    /// Set the `exampleMap` property in the response capnp builder to the rust map `data`
63    /// ```ignore
64    /// set_from_map(data, |size| response.init_example_map(size));
65    /// ```
66    pub(crate) fn set_map<'a, K, V, E, T>(data: T, init: impl FnOnce(u32) -> Builder<'a, E>)
67    where
68        T: IntoIterator<Item = (K, V)>,
69        T::IntoIter: ExactSizeIterator,
70        E: SetKeyValue<K, V>,
71    {
72        let data = data.into_iter();
73        let size = if let Ok(size) = data.len().try_into() {
74            size
75        } else {
76            panic!("map with more then u32::MAX entries")
77        };
78        let mut list = init(size);
79        for (i, (key, value)) in data.enumerate() {
80            let entry = list.reborrow().get(i as u32);
81            E::set_key_value(entry, key, value);
82        }
83    }
