50/// first usable file descriptor after stdin, stdout and stderr
51const FIRST_FD_AFTER_STDIO: RawFd = 3;

53impl ChildReaper {
54    pub fn get(&self, id: &str) -> Result<ReapableChild> {
55        let lock = lock!(self.grandchildren());
56        let r = lock
57            .get(id)
58            .and_then(|v| v.last())
59            .context("child not available")?
60            .clone();
61        Ok(r)
62    }

64    #[allow(clippy::too_many_arguments)]
65    pub async fn create_child<P, I, S>(
66        &self,
67        cmd: P,
68        args: I,
69        stdin: bool,
70        container_io: &mut ContainerIO,
71        pidfile: &Path,
72        env_vars: Vec<(String, String)>,
73        additional_fds: Vec<OwnedFd>,
74    ) -> Result<(u32, CancellationToken)>
75    where
76        P: AsRef<OsStr> + Debug,
77        I: IntoIterator<Item = S> + Debug,
78        S: AsRef<OsStr>,
79    {
80        debug!("Running: {:?} {:?}", cmd, args);
81        let mut cmd = Command::new(cmd);
82
83        if stdin {
84            cmd.stdin(Stdio::piped());
85        }
86
87        let mut child = cmd
88            .args(args)
89            .stdout(Stdio::piped())
90            .stderr(Stdio::piped())
91            .envs(env_vars)
92            .fd_mappings(
93                additional_fds
94                    .iter()
95                    .enumerate()
96                    .map(|(i, fd)| FdMapping {
97                        parent_fd: unsafe { OwnedFd::from_raw_fd(fd.as_raw_fd()) },
98                        child_fd: i as RawFd + FIRST_FD_AFTER_STDIO,
99                    })
100                    .collect(),
101            )?
102            .spawn()
103            .context("spawn child process: {}")?;
104
105        debug!(
106            "Running child on PID: {}",
107            child.id().map_or("unknown".into(), |x| x.to_string())
108        );
109
110        // close file descriptors after spawn
111        drop(additional_fds);
112
113        let token = CancellationToken::new();
114
115        match container_io.typ_mut() {
116            ContainerIOType::Terminal(terminal) => {
117                terminal
118                    .wait_connected(stdin, token.clone())
119                    .await
120                    .context("wait for terminal socket connection")?;
121            }
122            ContainerIOType::Streams(streams) => {
123                let stdout = child.stdout.take();
124                let stderr = child.stderr.take();
125                let stdin = child.stdin.take();
126                streams.handle_stdio_receive(stdin, stdout, stderr, token.clone());
127            }
128        };
129
130        let output = child.wait_with_output().await?;
131
132        if !output.status.success() {
133            const BASE_ERR: &str = "child command exited with";
134
135            let mut err_str = match output.status.code() {
136                Some(code) => format!("{BASE_ERR}: {code}"),
137                None => format!("{BASE_ERR} signal"),
138            };
139
140            if !output.stderr.is_empty() {
141                write!(
142                    err_str,
143                    ": {}",
144                    str::from_utf8(&output.stderr).context("convert stderr to utf8")?,
145                )?;
146            }
147            // token must be cancelled here because the child hasn't been registered yet,
148            // meaning there is no other entity that could cancel the read_loops.
149            token.cancel();
150
151            // Wait to ensure that all children do not become zombies.
152            Self::check_child_processes();
153
154            error!("Failed: {err_str}");
155            bail!(err_str)
156        }
157
158        let grandchild_pid = fs::read_to_string(pidfile)
159            .await
160            .with_context(|| format!("grandchild pid read error {}", pidfile.display()))?
161            .parse::<u32>()
162            .with_context(|| format!("grandchild pid parse error {}", pidfile.display()))?;
163
164        Ok((grandchild_pid, token))
165    }
166
167    fn check_child_processes() {
168        debug!("Checking child processes");
169        let pid = Pid::from_raw(-1);
170        loop {
171            match waitpid(pid, WaitPidFlag::WNOHANG.into()) {
172                Ok(WaitStatus::Exited(p, code)) => {
173                    debug!("PID {p} exited with status: {code}");
174                    break;
175                }
176                Ok(WaitStatus::StillAlive) => {
177                    debug!("PID {pid} is still in same state");
178                    break;
179                }
180                Ok(_) => {
181                    continue;
182                }
183                Err(Errno::EINTR) => {
184                    debug!("Retrying on EINTR for PID {pid}");
185                    continue;
186                }
187                Err(err) => {
188                    error!("Unable to waitpid on {:#}", err);
189                    break;
190                }
191            };
192        }
193    }

195    pub fn watch_grandchild(
196        &self,
197        child: Child,
198        leak_fds: Vec<OwnedFd>,
199    ) -> Result<Receiver<ExitChannelData>> {
200        let pid = child.pid;
201        let (id, mut reapable_grandchild) = ReapableChild::from_child(child);
202
203        // Create channels outside the lock to minimize contention
204        let (exit_tx, exit_rx) = reapable_grandchild.watch()?;
205
206        let locked_grandchildren = self.grandchildren();
207        {
208            let mut map = lock!(locked_grandchildren);
209            map.entry(id).or_default().push(reapable_grandchild);
210        }
211        let cleanup_grandchildren = locked_grandchildren.clone();
212
213        task::spawn(
214            async move {
215                exit_tx.subscribe().recv().await?;
216                drop(leak_fds);
217                Self::forget_grandchild(&cleanup_grandchildren, pid)
218            }
219            .instrument(debug_span!("watch_grandchild", pid)),
220        );
221        Ok(exit_rx)
222    }
223
224    fn forget_grandchild(
225        locked_grandchildren: &Arc<Mutex<HashMap<Box<str>, Vec<ReapableChild>>>>,
226        grandchild_pid: u32,
227    ) -> Result<()> {
228        let mut map = lock!(locked_grandchildren);
229        for children in map.values_mut() {
230            if let Some(pos) = children.iter().position(|v| v.pid == grandchild_pid) {
231                children.swap_remove(pos);
232                return Ok(());
233            }
234        }
235        Ok(())
236    }

238    pub async fn kill_grandchildren(&self, s: Signal) -> Result<()> {
239        debug!("Killing grandchildren");
240        // Collect only PIDs and task handles to avoid cloning entire ReapableChild
241        let children: Vec<_> = {
242            let map = lock!(self.grandchildren);
243            map.values()
244                .flatten()
245                .map(|gc| (gc.pid, gc.task.clone()))
246                .collect()
247        };
248        for (pid, task) in children {
249            let span = debug_span!("kill_grandchild", pid);
250            let _enter = span.enter();
251            debug!("Killing single grandchild");
252            kill_grandchild(pid, s);
253            if let Some(t) = task {
254                let handle = lock!(t).take();
255                if let Some(handle) = handle {
256                    let close_span = debug_span!("close", signal = s.as_str());
257                    async {
258                        debug!("Waiting for task to close");
259                        if let Err(e) = handle.await {
260                            error!("Unable to close grandchild: {:#}", e)
261                        }
262                    }
263                    .instrument(close_span)
264                    .await;
265                }
266            }
267            debug!("Done killing single grandchild");
268        }
269        debug!("Done killing all grandchildren");
270        Ok(())
271    }
272}

274pub fn kill_grandchild(raw_pid: u32, s: Signal) {
275    let pid = Pid::from_raw(raw_pid as pid_t);
276    if let Ok(pgid) = getpgid(Some(pid)) {
277        // If process_group is 1, we will end up calling
278        // kill(-1), which kills everything conmon is allowed to.
279        let pgid = i32::from(pgid);
280
281        #[allow(clippy::collapsible_if)]
282        if pgid > 1 {
283            if let Err(e) = kill(Pid::from_raw(-pgid), s) {
284                error!(
285                    raw_pid,
286                    "Failed to get pgid, falling back to killing pid: {:#}", e
287                );
288            }
289        }
290    }
291    if let Err(e) = kill(pid, s) {
292        debug!("Failed killing pid: {:#}", e);
293    }
294}

339    pub fn from_child(child: Child) -> (Box<str>, Self) {
340        (
341            child.id,
342            Self {
343                exit_paths: Arc::from(child.exit_paths),
344                oom_exit_paths: Arc::from(child.oom_exit_paths),
345                pid: child.pid,
346                io: child.io,
347                timeout: child.timeout,
348                token: child.token,
349                task: None,
350                cleanup_cmd: Arc::from(child.cleanup_cmd),
351            },
352        )

355    fn watch(&mut self) -> Result<(Sender<ExitChannelData>, Receiver<ExitChannelData>)> {
356        let exit_paths = self.exit_paths.clone();
357        let oom_exit_paths = self.oom_exit_paths.clone();
358        let pid = self.pid();
359        // Only one exit code will be written.
360        let (exit_tx, exit_rx) = broadcast::channel(1);
361        let exit_tx_clone = exit_tx.clone();
362        let timeout = *self.timeout();
363        let stop_token = self.token().clone();
364        let cleanup_cmd_raw = self.cleanup_cmd.clone();
365
366        let task = task::spawn(
367            async move {
368                debug!("Running task");
369                let mut exit_code: i32 = -1;
370                let mut oomed = false;
371                let mut timed_out = false;
372                let (oom_tx, mut oom_rx) = tokio::sync::mpsc::channel(1);
373                let oom_watcher = OOMWatcher::new(&stop_token, pid, &oom_exit_paths, oom_tx).await;
374
375                let span = debug_span!("wait_for_exit_code");
376                let wait_for_exit_code = task::spawn_blocking(move || {
377                    let _enter = span.enter();
378                    Self::wait_for_exit_code(&stop_token, pid)
379                });
380
381                let closure = async {
382                    let (code, oom) = tokio::join!(wait_for_exit_code, oom_rx.recv());
383                    if let Ok(code) = code {
384                        exit_code = code;
385                    }
386                    if let Some(event) = oom {
387                        oomed = event.oom;
388                    }
389                };
390                if let Some(timeout) = timeout {
391                    if time::timeout_at(timeout, closure).await.is_err() {
392                        timed_out = true;
393                        exit_code = -3;
394                        kill_grandchild(pid, Signal::SIGKILL);
395                    }
396                } else {
397                    closure.await;
398                }
399                oom_watcher.stop().await;
400
401                let exit_channel_data = ExitChannelData {
402                    exit_code,
403                    oomed,
404                    timed_out,
405                };
406                debug!(?exit_paths, "Write to exit paths");
407                if let Err(e) = Self::write_to_exit_paths(exit_code, &exit_paths).await {
408                    error!(pid, "Could not write exit paths: {:#}", e);
409                }
410
411                if !cleanup_cmd_raw.is_empty() {
412                    Self::spawn_cleanup_process(&cleanup_cmd_raw).await;
413                }
414
415                debug!("Sending exit struct to channel: {:?}", exit_channel_data);
416                if exit_tx_clone.send(exit_channel_data).is_err() {
417                    debug!("Unable to send exit status");
418                }
419                debug!("Task done");
420            }
421            .instrument(debug_span!("watch", pid)),
422        );
423
424        self.task = Some(Arc::new(Mutex::new(Some(task))));
425
426        Ok((exit_tx, exit_rx))
427    }

429    async fn spawn_cleanup_process(raw_cmd: &[String]) {
430        let mut cleanup_cmd = Command::new(&raw_cmd[0]);
431
432        cleanup_cmd.args(&raw_cmd[1..]);
433
434        tokio::spawn(async move {
435            match cleanup_cmd.status().await {
436                Ok(status) => {
437                    if !status.success() {
438                        error!("Failed to execute cleanup command successfully: {}", status);
439                    }
440                }
441                Err(e) => error!(
442                    "Failed to spawn and execute cleanup command process successfully: {}",
443                    e
444                ),
445            }
446        });
447    }
448
449    fn wait_for_exit_code(token: &CancellationToken, pid: u32) -> i32 {
450        debug!("Waiting for exit code");
451        const FAILED_EXIT_CODE: i32 = -3;
452        loop {
453            match waitpid(Pid::from_raw(pid as pid_t), None) {
454                Ok(WaitStatus::Exited(_, exit_code)) => {
455                    debug!("Exited {}", exit_code);
456                    token.cancel();
457                    return exit_code;
458                }
459                Ok(WaitStatus::Signaled(_, sig, _)) => {
460                    debug!("Signaled: {sig}");
461                    token.cancel();
462                    return (sig as i32) + 128;
463                }
464                Ok(_) => {
465                    continue;
466                }
467                Err(Errno::EINTR) => {
468                    debug!("Failed to wait for pid on EINTR, retrying");
469                    continue;
470                }
471                Err(err) => {
472                    error!("Unable to waitpid on {:#}", err);
473                    token.cancel();
474                    return FAILED_EXIT_CODE;
475                }
476            };
477        }
478    }

480    async fn write_to_exit_paths(code: i32, paths: &[PathBuf]) -> Result<()> {
481        let code_str = format!("{code}");
482        for path_buf in paths {
483            let span = debug_span!("write_exit_path", path = %path_buf.display());
484            async {
485                debug!("Creating exit file");
486                if let Ok(mut fp) = File::create(path_buf).await {
487                    debug!(code, "Writing exit code to file");
488                    if let Err(e) = fp.write_all(code_str.as_bytes()).await {
489                        error!("Could not write exit file to path: {:#}", e);
490                    }
491                    debug!("Flushing file");
492                    if let Err(e) = fp.flush().await {
493                        error!("Unable to flush {}: {:#}", path_buf.display(), e);
494                    }
495                    debug!("Done writing exit file");
496                }
497            }
498            .instrument(span)
499            .await;
500        }
501        Ok(())
502    }
