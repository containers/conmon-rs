68impl Terminal {
69    /// Setup a new terminal instance.
70    pub fn new(logger: SharedContainerLog, attach: SharedContainerAttach) -> Result<Self> {
71        debug!("Creating new terminal");
72        let path = ContainerIO::temp_file_name(None, "conmon-term-", ".sock")?;
73        let path_clone = path.clone();
74
75        let (ready_tx, ready_rx) = std::sync::mpsc::channel();
76        let (connected_tx, connected_rx) = mpsc::channel(1);
77
78        task::spawn(
79            async move {
80                if let Err(e) = Self::listen(Config {
81                    path: path_clone,
82                    ready_tx,
83                    connected_tx,
84                })
85                .await
86                {
87                    error!("Unable to listen on terminal: {:#}", e);
88                };
89            }
90            .instrument(debug_span!("listen")),
91        );
92        ready_rx.recv().context("wait for listener to be ready")?;
93
94        Ok(Self {
95            path,
96            connected_rx,
97            message_rx: None,
98            tty: None,
99            logger,
100            attach,
101        })
102    }
103
104    /// Waits for the socket client to be connected.
105    pub async fn wait_connected(&mut self, stdin: bool, token: CancellationToken) -> Result<()> {
106        debug!("Waiting for terminal socket connection");
107        let fd = self
108            .connected_rx
109            .recv()
110            .await
111            .context("receive connected channel")?;
112        let fd = Arc::new(TerminalFd::new(fd)?);
113        self.set_tty(Arc::downgrade(&fd).into());
114
115        debug!("Changing terminal settings");
116        let mut term = termios::tcgetattr(&fd)?;
117        term.output_flags |= OutputFlags::ONLCR;
118        termios::tcsetattr(&fd, SetArg::TCSANOW, &term)?;
119
120        let attach_clone = self.attach.clone();
121        let logger_clone = self.logger.clone();
122        let (message_tx, message_rx) = async_channel::bounded(10);
123        self.message_rx = Some(message_rx);
124
125        task::spawn({
126            let fd = fd.clone();
127            async move {
128                if let Err(e) = ContainerIO::read_loop(
129                    &*fd,
130                    Pipe::StdOut,
131                    logger_clone,
132                    message_tx,
133                    attach_clone,
134                )
135                .await
136                {
137                    error!("Stdout read loop failure: {:#}", e)
138                }
139                Ok::<_, anyhow::Error>(())
140            }
141            .instrument(debug_span!("read_loop"))
142        });
143
144        if stdin {
145            let attach_clone = self.attach.clone();
146            task::spawn(
147                async move {
148                    if let Err(e) = ContainerIO::read_loop_stdin(&*fd, attach_clone, token).await {
149                        error!("Stdin read loop failure: {:#}", e);
150                    }
151                }
152                .instrument(debug_span!("read_loop_stdin")),
153            );
154        }
155
156        Ok(())
157    }

159    /// Resize the terminal width and height.
160    pub fn resize(&self, width: u16, height: u16) -> Result<()> {
161        debug!("Resizing terminal to width {} and height {}", width, height);
162        let ws = winsize {
163            ws_row: height,
164            ws_col: width,
165            ws_xpixel: 0,
166            ws_ypixel: 0,
167        };
168        let tty = self.tty().as_ref().and_then(Weak::upgrade);
169        match unsafe {
170            libc::ioctl(
171                tty.context("terminal not connected")?.as_raw_fd(),
172                TIOCSWINSZ,
173                &ws,
174            )
175        } {
176            0 => Ok(()),
177            _ => Err(io::Error::last_os_error().into()),
178        }
179    }

181    async fn listen(config: Config) -> Result<()> {
182        let path = config.path();
183        debug!("Listening terminal socket on {}", path.display());
184        let listener = Listener::<DefaultListener>::default().bind_long_path(path)?;
185
186        // Update the permissions
187        let mut perms = fs::metadata(path).await?.permissions();
188        perms.set_mode(0o700);
189        fs::set_permissions(path, perms).await?;
190
191        config
192            .ready_tx()
193            .send(())
194            .map_err(|_| format_err!("unable to send ready message"))?;
195
196        let stream = listener.accept().await?.0;
197        debug!("Got terminal socket stream: {:?}", stream);
198
199        Self::handle_fd_receive(stream, config).await
200    }
201
202    async fn handle_fd_receive(mut stream: UnixStream, config: Config) -> Result<()> {
203        loop {
204            if !stream.ready(Interest::READABLE).await?.is_readable() {
205                continue;
206            }
207
208            let mut data_buffer = [];
209            let mut fd_buffer: [RawFd; 1] = [0];
210
211            match stream.recv_with_fd(&mut data_buffer, &mut fd_buffer) {
212                Ok((_, fd_read)) => {
213                    // take ownership of the received file descriptor (prevents fd leak in case of error)
214                    let fd = (fd_read != 0).then(|| unsafe { OwnedFd::from_raw_fd(fd_buffer[0]) });
215
216                    // Allow only one single read
217                    let path = config.path();
218                    debug!("Removing socket path {}", path.display());
219                    fs::remove_file(path).await?;
220
221                    debug!("Shutting down receiver stream");
222                    stream.shutdown().await?;
223
224                    if fd.is_none() {
225                        error!("No file descriptor received");
226                    }
227
228                    let fd = fd.context("got no file descriptor")?;
229
230                    debug!("Received terminal file descriptor");
231
232                    config
233                        .connected_tx
234                        .send(fd)
235                        .await
236                        .context("send connected channel")?;
237
238                    debug!("Shutting down listener thread");
239                    return Ok(());
240                }
241                Err(ref e) if e.kind() == ErrorKind::WouldBlock => {
242                    trace!("WouldBlock error, retrying");
243                    continue;
244                }
245                Err(e) => {
246                    error!("Unable to receive data: {}", e);
247                    return Err(e.into());
248                }
249            }
250        }
251    }
252}

254impl Drop for Terminal {
255    fn drop(&mut self) {
256        if let Err(e) = std::fs::remove_file(self.path()) {
257            trace!(
258                "Unable to remove socket file path {}: {}",
259                self.path().display(),
260                e
261            )
262        }
263    }

275        let logger = ContainerLog::new();
276        let attach = SharedContainerAttach::default();
277        let token = CancellationToken::new();
278
279        let mut sut = Terminal::new(logger, attach)?;
280        assert!(sut.path().exists());
281
282        let res = pty::openpty(None, None)?;
283
284        let stream = UnixStream::connect(sut.path()).await?;
285        loop {
286            let ready = stream.ready(Interest::WRITABLE).await?;
287            if ready.is_writable() {
288                match stream.send_with_fd(b"test", &[res.master.as_raw_fd()]) {
289                    Ok(_) => break,
290                    Err(ref e) if e.kind() == ErrorKind::WouldBlock => continue,
291                    Err(e) => anyhow::bail!(e),
292                }
293            }
294        }
295
296        sut.wait_connected(true, token).await?;
297        assert!(!sut.path().exists());
298
299        // Write to the slave
300        let mut file: std::fs::File = res.slave.into();
301        file.write_all(b"test")?;
302
303        Ok(())
304    }

307#[derive(Debug)]
308struct TerminalFd(AsyncFd<std::fs::File>);
309
310impl TerminalFd {
311    fn new(fd: OwnedFd) -> io::Result<Self> {
312        let flags = fcntl::fcntl(&fd, FcntlArg::F_GETFL)?;
313        let flags = OFlag::from_bits_truncate(flags) | OFlag::O_NONBLOCK;
314        fcntl::fcntl(&fd, FcntlArg::F_SETFL(flags))?;
315        AsyncFd::new(fd.into()).map(Self)
316    }
317}
318
319impl AsRawFd for TerminalFd {

321        self.0.as_raw_fd()
322    }

331impl AsyncRead for &TerminalFd {
332    fn poll_read(
333        self: Pin<&mut Self>,
334        cx: &mut Context,
335        buf: &mut ReadBuf,
336    ) -> Poll<io::Result<()>> {
337        loop {
338            let mut guard = ready!(self.0.poll_read_ready(cx))?;
339            match guard.try_io(|inner| inner.get_ref().read(buf.initialize_unfilled())) {
340                Ok(n) => {
341                    buf.advance(n?);
342                    break Poll::Ready(Ok(()));
343                }
344                Err(_would_block) => continue,
345            }
346        }
347    }
348}

350impl AsyncWrite for &TerminalFd {
351    fn poll_write(self: Pin<&mut Self>, cx: &mut Context, buf: &[u8]) -> Poll<io::Result<usize>> {
352        loop {
353            let mut guard = ready!(self.0.poll_write_ready(cx))?;
354            match guard.try_io(|inner| inner.get_ref().write(buf)) {
355                Ok(result) => break Poll::Ready(result),
356                Err(_would_block) => continue,
357            }
358        }
359    }
360
361    fn poll_flush(self: Pin<&mut Self>, _cx: &mut Context) -> Poll<io::Result<()>> {
362        Poll::Ready(Ok(()))
363    }
364
365    fn poll_shutdown(self: Pin<&mut Self>, _cx: &mut Context) -> Poll<io::Result<()>> {
366        Poll::Ready(Ok(()))
367    }
