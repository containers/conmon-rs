82impl Terminal {
83    /// Setup a new terminal instance.
84    pub fn new(logger: SharedContainerLog, attach: SharedContainerAttach) -> Result<Self> {
85        debug!("Creating new terminal");
86        let path = ContainerIO::temp_file_name(None, "conmon-term-", ".sock");
87        let path_clone = path.clone();
88
89        let (ready_tx, ready_rx) = std::sync::mpsc::channel();
90        let (connected_tx, connected_rx) = mpsc::channel(10);
91
92        task::spawn(
93            async move {
94                if let Err(e) = Self::listen(Config {
95                    path: path_clone,
96                    ready_tx,
97                    connected_tx,
98                })
99                .await
100                {
101                    error!("Unable to listen on terminal: {:#}", e);
102                };
103            }
104            .instrument(debug_span!("listen")),
105        );
106        ready_rx.recv().context("wait for listener to be ready")?;
107
108        Ok(Self {
109            path,
110            connected_rx,
111            message_rx: None,
112            tty: None,
113            logger,
114            attach,
115        })
116    }
117
118    /// Waits for the socket client to be connected.
119    pub async fn wait_connected(&mut self, stdin: bool, token: CancellationToken) -> Result<()> {
120        debug!("Waiting for terminal socket connection");
121        let fd = self
122            .connected_rx
123            .recv()
124            .await
125            .context("receive connected channel")?;
126        let fd = Arc::new(TerminalFd::new(fd)?);
127        self.set_tty(Arc::downgrade(&fd).into());
128
129        debug!("Changing terminal settings");
130        let mut term = termios::tcgetattr(&fd)?;
131        term.output_flags |= OutputFlags::ONLCR;
132        termios::tcsetattr(&fd, SetArg::TCSANOW, &term)?;
133
134        let attach_clone = self.attach.clone();
135        let logger_clone = self.logger.clone();
136        let (message_tx, message_rx) = async_channel::bounded(100);
137        self.message_rx = Some(message_rx);
138
139        task::spawn({
140            let fd = fd.clone();
141            async move {
142                if let Err(e) = ContainerIO::read_loop(
143                    &*fd,
144                    Pipe::StdOut,
145                    logger_clone,
146                    message_tx,
147                    attach_clone,
148                )
149                .await
150                {
151                    error!("Stdout read loop failure: {:#}", e)
152                }
153                Ok::<_, anyhow::Error>(())
154            }
155            .instrument(debug_span!("read_loop"))
156        });
157
158        if stdin {
159            let attach_clone = self.attach.clone();
160            task::spawn(
161                async move {
162                    if let Err(e) = ContainerIO::read_loop_stdin(&*fd, attach_clone, token).await {
163                        error!("Stdin read loop failure: {:#}", e);
164                    }
165                }
166                .instrument(debug_span!("read_loop_stdin")),
167            );
168        }
169
170        Ok(())
171    }

173    /// Resize the terminal width and height.
174    pub fn resize(&self, width: u16, height: u16) -> Result<()> {
175        debug!("Resizing terminal to width {} and height {}", width, height);
176        let ws = winsize {
177            ws_row: height,
178            ws_col: width,
179            ws_xpixel: 0,
180            ws_ypixel: 0,
181        };
182        let tty = self.tty().as_ref().and_then(Weak::upgrade);
183        match unsafe {
184            libc::ioctl(
185                tty.context("terminal not connected")?.as_raw_fd(),
186                TIOCSWINSZ,
187                &ws,
188            )
189        } {
190            0 => Ok(()),
191            _ => Err(io::Error::last_os_error().into()),
192        }
193    }

195    async fn listen(config: Config) -> Result<()> {
196        let path = config.path();
197        debug!("Listening terminal socket on {}", path.display());
198        let listener = Listener::<DefaultListener>::default().bind_long_path(path)?;
199
200        // Update the permissions
201        let mut perms = fs::metadata(path).await?.permissions();
202        perms.set_mode(0o700);
203        fs::set_permissions(path, perms).await?;
204
205        config
206            .ready_tx()
207            .send(())
208            .map_err(|_| format_err!("unable to send ready message"))?;
209
210        let stream = listener.accept().await?.0;
211        debug!("Got terminal socket stream: {:?}", stream);
212
213        Self::handle_fd_receive(stream, config).await
214    }
215
216    async fn handle_fd_receive(mut stream: UnixStream, config: Config) -> Result<()> {
217        loop {
218            if !stream.ready(Interest::READABLE).await?.is_readable() {
219                continue;
220            }
221
222            let mut data_buffer = [];
223            let mut fd_buffer: [RawFd; 1] = [0];
224
225            match stream.recv_with_fd(&mut data_buffer, &mut fd_buffer) {
226                Ok((_, fd_read)) => {
227                    // take ownership of the received file descriptor (prevents fd leak in case of error)
228                    let fd = (fd_read != 0).then(|| unsafe { OwnedFd::from_raw_fd(fd_buffer[0]) });
229
230                    // Allow only one single read
231                    let path = config.path();
232                    debug!("Removing socket path {}", path.display());
233                    fs::remove_file(path).await?;
234
235                    debug!("Shutting down receiver stream");
236                    stream.shutdown().await?;
237
238                    if fd.is_none() {
239                        error!("No file descriptor received");
240                    }
241
242                    let fd = fd.context("got no file descriptor")?;
243
244                    debug!("Received terminal file descriptor");
245
246                    config
247                        .connected_tx
248                        .send(fd)
249                        .await
250                        .context("send connected channel")?;
251
252                    debug!("Shutting down listener thread");
253                    return Ok(());
254                }
255                Err(ref e) if e.kind() == ErrorKind::WouldBlock => {
256                    trace!("WouldBlock error, retrying");
257                    continue;
258                }
259                Err(e) => {
260                    error!("Unable to receive data: {}", e);
261                    return Err(e.into());
262                }
263            }
264        }
265    }
266}

268impl Drop for Terminal {
269    fn drop(&mut self) {
270        if let Err(e) = std::fs::remove_file(self.path()) {
271            trace!(
272                "Unable to remove socket file path {}: {}",
273                self.path().display(),
274                e
275            )
276        }
277    }

289        let logger = ContainerLog::new();

291        let token = CancellationToken::new();
292

294        assert!(sut.path().exists());

296        let res = pty::openpty(None, None)?;
297

299        loop {
300            let ready = stream.ready(Interest::WRITABLE).await?;
301            if ready.is_writable() {
302                match stream.send_with_fd(b"test", &[res.master.as_raw_fd()]) {
303                    Ok(_) => break,
304                    Err(ref e) if e.kind() == ErrorKind::WouldBlock => continue,
305                    Err(e) => anyhow::bail!(e),
306                }
307            }
308        }
309
310        sut.wait_connected(true, token).await?;
311        assert!(!sut.path().exists());
312
313        // Write to the slave
314        let mut file: std::fs::File = res.slave.into();
315        file.write_all(b"test")?;
316
317        Ok(())
318    }

321#[derive(Debug)]
322struct TerminalFd(AsyncFd<std::fs::File>);
323
324impl TerminalFd {
325    fn new(fd: OwnedFd) -> io::Result<Self> {
326        let flags = fcntl::fcntl(&fd, FcntlArg::F_GETFL)?;
327        let flags = OFlag::from_bits_truncate(flags) | OFlag::O_NONBLOCK;
328        fcntl::fcntl(&fd, FcntlArg::F_SETFL(flags))?;
329        AsyncFd::new(fd.into()).map(Self)
330    }
331}

345impl AsyncRead for &TerminalFd {
346    fn poll_read(
347        self: Pin<&mut Self>,
348        cx: &mut Context,
349        buf: &mut ReadBuf,
350    ) -> Poll<io::Result<()>> {
351        loop {
352            let mut guard = ready!(self.0.poll_read_ready(cx))?;
353            match guard.try_io(|inner| inner.get_ref().read(buf.initialize_unfilled())) {
354                Ok(n) => {
355                    buf.advance(n?);
356                    break Poll::Ready(Ok(()));
357                }
358                Err(_would_block) => continue,
359            }
360        }
361    }
362}
363
364impl AsyncWrite for &TerminalFd {
365    fn poll_write(self: Pin<&mut Self>, cx: &mut Context, buf: &[u8]) -> Poll<io::Result<usize>> {
366        loop {
367            let mut guard = ready!(self.0.poll_write_ready(cx))?;
368            match guard.try_io(|inner| inner.get_ref().write(buf)) {
369                Ok(result) => break Poll::Ready(result),
370                Err(_would_block) => continue,
371            }
372        }
373    }
374
375    fn poll_flush(self: Pin<&mut Self>, _cx: &mut Context) -> Poll<io::Result<()>> {
376        Poll::Ready(Ok(()))
377    }
378
379    fn poll_shutdown(self: Pin<&mut Self>, _cx: &mut Context) -> Poll<io::Result<()>> {
380        Poll::Ready(Ok(()))
381    }
